// iterative for loop
      for i in range(10):
         print(i) // will iteratively output all numbers from 0 to 9 inclusive

      // for each loop
      my_array = [1, 3, 5, 6, 4 ,2]
      for i in my_array:
         print(i) // will print all elements in my_array starting from 0th index

      // while loop
      i = 0
      while i  < 10:
         print(i) //prints out all numbers from 0 to 9 inclusive
         i = i + 1
