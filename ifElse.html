      if a == 1:

      # if else
      if a == 1:
            # a is 1
      else:
            # a isn't 1
---------------------------------------
      # if / else/ elif
      if a == 1:
            # a is 1
      elif b == 1:
            # b is 1
      else:
            # a isn't 1 and b isn't 1
---------------------------------------
      # multiple if
      if a == 1:
         # a is 1
      if b == 1:
         # b is 1
