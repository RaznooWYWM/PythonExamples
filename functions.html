#returns a boolean value determining if the passed value is greater than 2
      def greaterThanTwo(x):
            result = x > 2
            return result
      
      # adds 3 to the passed number
      def addThree(x):
            result = x + 3
            return result
      
      # returns a list of integers from x to 0 or an empty list if x <= 0
      def makeList(x):
          resultList = ()
          while x > 0:
               resultList.append(x)
               x = x - 1
          return resultList
      
      greaterThanTwo(1)    #False
      greaterThanTwo(30)   #True
      
      addThree(2)  # 5
      addThree(0)  # 3
      
      makeList(5) # (5, 4, 3, 2, 1)
      makeList(0) # ()
