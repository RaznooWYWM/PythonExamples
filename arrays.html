// python does not have arrays but the list syntax is similar to arrays

      // Lists
      my_list = ()    //starts as an empty list
      my_list += 3    // add the integer 3 to my_list
      my_list += 5    // add the integer 5 to my_list
      my_list         // equivalent to (3, 5)
      my_list[1]      // equivalent to 5
      my_list[1] = 7  //reassign the contents of the 1 index
      my_list         // equivalent to (3, 7)
